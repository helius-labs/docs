  <Card title="getAsset" href="/zh/api-reference/das/getasset">
    根据资产 ID 检索特定资产的信息。
  </Card>
  <Card title="getAssetBatch" href="/zh/api-reference/das/getassetbatch">
    在单个请求中检索多个资产的信息。
  </Card>
  <Card title="getAssetProof" href="/zh/api-reference/das/getassetproof">
    获取特定压缩资产的证明。
  </Card>
  <Card title="getAssetProofBatch" href="/zh/api-reference/das/getassetproofbatch">
    在单个请求中获取多个压缩资产的证明。
  </Card>
  <Card title="getAssetsByAuthority" href="/zh/api-reference/das/getassetsbyauthority">
    根据指定的权限地址返回资产。
  </Card>
  <Card title="getAssetsByCreator" href="/zh/api-reference/das/getassetsbycreator">
    检索由特定创建者地址创建的资产。
  </Card>
  <Card title="getAssetsByGroup" href="/zh/api-reference/das/getassetsbygroup">
    返回属于特定组的资产。
  </Card>
  <Card title="getAssetsByOwner" href="/zh/api-reference/das/getassetsbyowner">
    列出由特定地址拥有的所有资产。
  </Card>
  <Card title="getNftEditions" href="/zh/api-reference/das/getnfteditions">
    检索有关 NFT 版本的信息。
  </Card>
  <Card title="getSignaturesForAsset" href="/zh/api-reference/das/getsignaturesforasset">
    返回涉及特定资产的交易签名。
  </Card>
  <Card title="getTokenAccounts" href="/zh/api-reference/das/gettokenaccounts">
    根据指定参数检索代币账户。
  </Card>
  <Card title="searchAssets" href="/zh/api-reference/das/searchassets">
    根据各种标准搜索资产。
  </Card>
