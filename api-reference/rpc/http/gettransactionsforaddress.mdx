## Request Parameters

<ParamField path="params[0]" type="string" required>
Solana account address to retrieve transaction history for (wallet, token, program, NFT, etc.).
</ParamField>

<ParamField path="params[1]" type="object">
Advanced query configuration for filtering, sorting, and pagination.

<Expandable title="Basic Parameters">
  <ParamField path="transactionDetails" type="string" default="signatures">
  Level of transaction detail to return.
  
  - `"signatures"` - Return only signature metadata (default)
  - `"full"` - Return complete transaction data
  </ParamField>
  
  <ParamField path="limit" type="integer" default="1000">
  Maximum number of transactions to return in a single request. Range: 1-1,000.
  </ParamField>
  
  <ParamField path="sortOrder" type="string" default="desc">
  Sort order for returned transactions.
  - `"desc"` - Newest transactions first (default)
  - `"asc"` - Oldest transactions first
  </ParamField>
  
  <ParamField path="paginationToken" type="string">
  Pagination token from previous response to get next page of results. Format: `"slot:position"`.
  </ParamField>
</Expandable>

<Expandable title="Commitment & Context">
  <ParamField path="commitment" type="string" default="finalized">
  The commitment level for the request.
  - `"processed"` - The node has seen the transaction
  - `"confirmed"` - The transaction has been confirmed by the cluster
  - `"finalized"` - The transaction has been finalized by the cluster (default)
  </ParamField>
  
  <ParamField path="minContextSlot" type="integer">
  Minimum context slot to use for request (optional).
  </ParamField>
</Expandable>

<Expandable title="Full Transaction Parameters">
  <ParamField path="encoding" type="string" default="json">
  Encoding format for transaction data (applies only when `transactionDetails="full"`).
  
  - `"json"` - JSON encoding (default)
  - `"jsonParsed"` - Parsed JSON encoding
  - `"base58"` - Base58 encoding
  - `"base64"` - Base64 encoding
  </ParamField>
  
  <ParamField path="maxSupportedTransactionVersion" type="integer">
  Maximum transaction version to return (applies only when `transactionDetails="full"`).
  </ParamField>
</Expandable>

<Expandable title="Filter Options">
  <ParamField path="filters" type="object">
  Advanced filters to narrow down transaction results.
  
  <Expandable title="Slot Filter">
    <ParamField path="filters.slot" type="object">
    Filter transactions by slot number. Supports multiple comparison operators.
    
    <ParamField path="filters.slot.gte" type="integer">
    Include transactions with slot greater than or equal to this value.
    </ParamField>
    
    <ParamField path="filters.slot.gt" type="integer">
    Include transactions with slot greater than this value.
    </ParamField>
    
    <ParamField path="filters.slot.lte" type="integer">
    Include transactions with slot less than or equal to this value.
    </ParamField>
    
    <ParamField path="filters.slot.lt" type="integer">
    Include transactions with slot less than this value.
    </ParamField>
    
    <ParamField path="filters.slot.eq" type="integer">
    Include transactions with slot equal to this value.
    </ParamField>
    </ParamField>
  </Expandable>
  
  <Expandable title="Block Time Filter">
    <ParamField path="filters.blockTime" type="object">
    Filter transactions by block timestamp (Unix timestamp). Supports multiple comparison operators.
    
    <ParamField path="filters.blockTime.gte" type="integer">
    Include transactions with blockTime greater than or equal to this Unix timestamp.
    </ParamField>
    
    <ParamField path="filters.blockTime.gt" type="integer">
    Include transactions with blockTime greater than this Unix timestamp.
    </ParamField>
    
    <ParamField path="filters.blockTime.lte" type="integer">
    Include transactions with blockTime less than or equal to this Unix timestamp.
    </ParamField>
    
    <ParamField path="filters.blockTime.lt" type="integer">
    Include transactions with blockTime less than this Unix timestamp.
    </ParamField>
    
    <ParamField path="filters.blockTime.eq" type="integer">
    Include transactions with blockTime equal to this Unix timestamp.
    </ParamField>
    </ParamField>
  </Expandable>
  
  <Expandable title="Signature Filter">
    <ParamField path="filters.signature" type="object">
    Filter transactions by transaction signature for pagination purposes.
    
    <ParamField path="filters.signature.lt" type="string">
    Get transactions before this signature (earlier in time).
    </ParamField>
    
    <ParamField path="filters.signature.gt" type="string">
    Get transactions after this signature (later in time).
    </ParamField>
    </ParamField>
  </Expandable>
  
  <Expandable title="Status Filter">
    <ParamField path="filters.status" type="string" default="any">
    Filter transactions by success or failure status.
    
    - `"succeeded"` - Include only successful transactions
    - `"failed"` - Include only failed transactions
    - `"any"` - Include all transactions regardless of status (default)
    </ParamField>
  </Expandable>
  </ParamField>
</Expandable>
</ParamField>

## Response Fields

<ResponseField name="data" type="array" required>
List of transaction information. Format depends on `transactionDetails` parameter.

<Expandable title="Signature Format (transactionDetails='signatures')">
  <ResponseField name="signature" type="string" required>
  Transaction signature as a base-58 encoded string.
  </ResponseField>
  
  <ResponseField name="slot" type="integer" required>
  The slot that contains the block with the transaction.
  </ResponseField>
  
  <ResponseField name="err" type="object | null">
  Error object if the transaction failed, or null if successful.
  </ResponseField>
  
  <ResponseField name="memo" type="string | null">
  Memo associated with the transaction, or null if none.
  </ResponseField>
  
  <ResponseField name="blockTime" type="integer | null">
  Estimated production time as Unix timestamp (seconds since epoch), or null if not available.
  </ResponseField>
  
  <ResponseField name="confirmationStatus" type="string | null">
  Transaction's cluster confirmation status. Values: `"processed"`, `"confirmed"`, or `"finalized"`.
  </ResponseField>
</Expandable>

<Expandable title="Full Transaction Format (transactionDetails='full')">
  <ResponseField name="signature" type="string" required>
  Transaction signature as a base-58 encoded string.
  </ResponseField>

  <ResponseField name="slot" type="integer" required>
  Slot that processed the transaction.
  </ResponseField>

  <ResponseField name="blockTime" type="integer | null">
  Estimated production time as Unix timestamp (seconds since epoch), or null if not available.
  </ResponseField>

  <ResponseField name="transaction" type="object" required>
  Complete transaction payload matching the `getTransaction` response.

  <Expandable title="transaction.signatures">
    <ResponseField name="transaction.signatures[]" type="string" required>
    Cryptographic signatures that authorize the transaction.
    </ResponseField>
  </Expandable>

  <Expandable title="transaction.message">
    <ResponseField name="transaction.message.accountKeys[]" type="string" required>
    Ordered list of account public keys referenced by the transaction.
    </ResponseField>

    <ResponseField name="transaction.message.header" type="object" required>
    Transaction header defining signer and read-only account counts.
    </ResponseField>

    <ResponseField name="transaction.message.instructions[]" type="object" required>
    Program instructions executed by the transaction. Each entry exposes `programIdIndex`, `accounts`, and `data` fields.
    </ResponseField>

    <ResponseField name="transaction.message.recentBlockhash" type="string" required>
    Recent blockhash that anchors the transactionâ€™s validity window.
    </ResponseField>
  </Expandable>
  </ResponseField>

  <ResponseField name="meta" type="object | null">
  Execution metadata returned by `getTransaction`. Includes:
  - `err`: error object when execution fails (null on success)
  - `fee`: lamports paid for the transaction
  - `logMessages`: array of program log strings (when available)
  - `preBalances` / `postBalances`: lamport balances for each account before and after execution
  - `preTokenBalances` / `postTokenBalances`: SPL token balances before and after execution
  - `innerInstructions`: nested program instructions emitted during execution
  - `loadedAddresses`: lookup table accounts (for versioned transactions)
  - `rewards`: validator reward information when present
  </ResponseField>
</Expandable>
</ResponseField>

<ResponseField name="paginationToken" type="string | null">
Pagination token for next page, or null if no more results. Format: `"slot:position"`.
</ResponseField>
