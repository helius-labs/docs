<CardGroup cols={2}>
  <Card title="Parse Transactions" href="/enhanced-transactions-api/gettransactions">
    Parses and returns enhanced, human-readable versions of the given transactions.
  </Card>
  <Card title="Get Transactions By Address" href="/enhanced-transactions-api/gettransactionsbyaddress">
    Retrieves enhanced transaction history for a specific address.
  </Card>
  <Card title="Get Raw Transactions By Address" href="/enhanced-transactions-api/getrawtransactionsbyaddress">
    Returns raw transaction history for a given address.
  </Card>
  <Card title="Get Raw Transactions" href="/enhanced-transactions-api/getrawtransactions">
    Returns an array of raw transactions from RPC.
  </Card>
  <Card title="Query Raw Transactions" href="/enhanced-transactions-api/queryrawtransactions">
    Returns an array of raw transactions for the given accounts with additional query options.
  </Card>
</CardGroup>
